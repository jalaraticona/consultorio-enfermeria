create table usuarios(
    id_usuario int not null AUTO_INCREMENT PRIMARY KEY,
    user varchar(50),
    password varchar(50),
    tipo varchar(50),
    estado varchar(20),
    id_enfermera int,
    FOREIGN KEY (id_enfermera) REFERENCES enfermera(id_enfermera)
)

create table enfermera(
    id_enfermera int not null AUTO_INCREMENT PRIMARY KEY,
    nombre varchar(50),
    paterno varchar(50),
    materno varchar(50),
    fec_nac date,
    ci int,
    expedido varchar(20),
    sexo varchar(20),
    fec_reg date
)

create table paciente(
    id_paciente int not null AUTO_INCREMENT PRIMARY KEY,
    nombre varchar(50),
    paterno varchar(50),
    materno varchar(50),
    fec_nac date,
	ci int,
    expedido varchar(20),
    sexo varchar(20),
    fec_reg date,
    residencia varchar(50),
    categoria varchar(50),
    carrera_cargo varchar(50)
)

create table insumos(
    id_insumo int not null AUTO_INCREMENT PRIMARY KEY,
    nombre varchar(100),
    medida varchar(100),
    tipo varchar(50),
    fec_ing date,
    fec_exp date,
    stock int,
    cant_disp int,
    comprobante int,
    lote varchar(50),
    origen varchar(100),
    red varchar(100),
    estado varchar(20),
    id_enfermera int,
    FOREIGN KEY (id_enfermera) REFERENCES enfermera(id_enfermera)
)

create table servicio(
	id_servicio int not null AUTO_INCREMENT PRIMARY KEY,
	nombre varchar(100),
	nro_dosis int,
	detalle text,
	costo float,
	tipo varchar(50)
)

create table registrahistoria(
    id_historia int not null AUTO_INCREMENT PRIMARY KEY,
    motivo text,
    fec_reg date,
    lugar varchar(100),
    dosis varchar(50),
    id_enfermera int,
    id_paciente int,
    id_servicio int,
    FOREIGN KEY (id_enfermera) REFERENCES enfermera(id_enfermera),
    FOREIGN KEY (id_paciente) REFERENCES paciente(id_paciente),
    FOREIGN KEY (id_servicio) REFERENCES servicio(id_servicio)
)

create table registrodiario(
    id_reg_diario int not null AUTO_INCREMENT PRIMARY KEY,
    fec_registro date,
    sal_ant int,
    cant_ing int,
    cant_disp int,
    cant_egre int,
    ori_dest varchar(100),
    fa int,
    fr int,
    fm int,
    fe int,
    cant_per int,
    saldo int,
    id_enfermera int,
    id_insumo int,
    FOREIGN KEY (id_enfermera) REFERENCES enfermera(id_enfermera),
    FOREIGN KEY (id_insumo) REFERENCES insumos(id_insumo)
)
